19function q  = runMadwickUpdate(qIn,Accelerometer,Gyroscope,Magnetometer,Beta)
20q=qIn;
21SamplePeriod=single(5/1000);
22
23% Normalise accelerometer measurement
24Accelerometer = Accelerometer / norm(Accelerometer);    % normalise magnitude
25% Normalise magnetometer measurement
26Magnetometer = Magnetometer / norm(Magnetometer);   % normalise magnitude
27% Reference direction of Earth's magnetic feild
28h = quaternProd(q, quaternProd([0 Magnetometer], quaternConj(q)));
29b = [0 norm([h(2) h(3)]) 0 h(4)];
30% Gradient decent algorithm corrective step
31F = [2*(q(2)*q(4) - q(1)*q(3)) - Accelerometer(1)
32    2*(q(1)*q(2) + q(3)*q(4)) - Accelerometer(2)
33    2*(0.5 - q(2)^2 - q(3)^2) - Accelerometer(3)
34    2*b(2)*(0.5 - q(3)^2 - q(4)^2) + 2*b(4)*(q(2)*q(4) - q(1)*q(3)) - Magnetometer(1)
35    2*b(2)*(q(2)*q(3) - q(1)*q(4)) + 2*b(4)*(q(1)*q(2) + q(3)*q(4)) - Magnetometer(2)
36    2*b(2)*(q(1)*q(3) + q(2)*q(4)) + 2*b(4)*(0.5 - q(2)^2 - q(3)^2) - Magnetometer(3)];
37J = [-2*q(3),                   2*q(4),                    -2*q(1),                         2*q(2)
38    2*q(2),                     2*q(1),                     2*q(4),                         2*q(3)
39    0,                         -4*q(2),                    -4*q(3),                         0
40    -2*b(4)*q(3),               2*b(4)*q(4),               -4*b(2)*q(3)-2*b(4)*q(1),       -4*b(2)*q(4)+2*b(4)*q(2)
41    -2*b(2)*q(4)+2*b(4)*q(2),   2*b(2)*q(3)+2*b(4)*q(1),    2*b(2)*q(2)+2*b(4)*q(4),       -2*b(2)*q(1)+2*b(4)*q(3)
42    2*b(2)*q(3),                2*b(2)*q(4)-4*b(4)*q(2),    2*b(2)*q(1)-4*b(4)*q(3),        2*b(2)*q(2)];
43step = (J'*F);
44step = step / norm(step);   % normalise step magnitude
45
46% Compute rate of change of quaternion
47qDot = 0.5 * quaternProd(q, [0 Gyroscope(1) Gyroscope(2) Gyroscope(3)]) - Beta * step';
48
49% Integrate to yield quaternion
50q = q + qDot * SamplePeriod;
51q = q / norm(q); % normalise quaternion
52
